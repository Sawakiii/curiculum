4. サーバーの再起動を自動化する 9220dd

現状では、ファイルを書き換える度に、サーバーを一時停止([Ctrl+c])・再起動しなければ、ブラウザをリロードしても変更が反映されません。
nodemonを導入して自動化します。

$ yarn add nodemon

npm scriptを編集します。

"scripts": {
    "start": "nodemon ./bin/www"
}

$ yarn start

これで自動化されました。

6. ルーティングを知ろう。

前章にて、app.jsにて説明を省いた部分がありました。app.jsには、以下のようなコードがあるはずです。

var indexRouter = require('./routes/index');
var usersRouter = require('./routes/users');
.
.
.
app.use('/', indexRouter);
app.use('/users', usersRouter);

ここでは、ルーティング(経路)に関するセットアップを行っています。
ユーザーが特定のURLにアクセスした際の処理を、アプリケーションは予め設定しておく必要があります(設定することで、URLに応じて、異なる処理を行うことが出来ます)。

app.use('/', indexRouter)は、リクエストURLが/(例: http://tkmtz.com/)だった場合の処理を、indexRouterに委譲する処理です。
同じく、app.use('/users', usersRouter)は、リクエストURLが/users(例:http://tkmtz.com/users/)以下の処理を、usersRouterに委譲します。

usersRouterもとい、routes/users.jsを見てみましょう。

var express = require('express');
var router = express.Router();

/* GET users listing. */
router.get('/', function(req, res, next) {
  res.send('respond with a resource');
});

module.exports = router;

このファイルでは、リクエストURLが/users、更に、/に合致する際の処理を定義しています。つまり、/users/に合致する際の処理です。

/users/にgetメソッドでリクエストした際に、res.send()を用いて文字列を返却しています。

以下にアクセスしてみましょう。
http://localhost:3000/users/
respond with a resourceが表示されるはずです。

7. データベースのセットアップ

単純な静的ページを提供するだけのwebサイトであれば、情報の保存にそこまで気を配る必要はありません。

しかし、Webアプリケーションを作成するには、多様なデータを効率よく保存・操作する必要があります。
そこで、データベースと呼ばれるソフトウェアを使用します。

expressは、標準のデータベースが用意されているわけではありません。

世の中で使用されているデータベースは、RDBMSと呼ばれるものが殆どです。
通常はSQL言語を使用してデータベースに命令を下します。
今回は、mongodbというデータベースソフトを使用します。
mongodbはSQL言語を使用しないため、NoSQLと呼ばれています。

const express = require('express')
const router = express.Router()
const ObjectID = require('mongodb').ObjectID
// MongoDB用ファイルを指定
const collection = require('../mongo')
const collectionName = 'api'

// (授業にて解説します。)
router.all('/*', function (req, res, next) {
    res.contentType('json')
    res.header('Access-Control-Allow-Origin', '*')
    next()
} );

// localhost:3000/api/にアクセスした際の処理を作成する
router.get('/', function (req, res) {
    collection(collectionName).find().toArray(function(err, docs){
        res.send(docs)
    })
});

// localhost:3000/api/(object id)/にアクセスした際の処理を記述します。
router.get('/:id', function (req, res) {
    collection(collectionName).findOne({_id: new ObjectID( req.params.id)}, {}, function(err, r){
        res.send(r);
    });
});

// postメソッドを使用します。postとは一体何でしょうか、添付のurlを参考に調べてみましょう。
router.post('/', function (req, res) {
    collection(collectionName).insertOne(req.body).then(function(r) {
        res.send(r)
    });
});

// putメソッドを使用します。putとは一体何でしょうか、添付のurlを参考に調べてみましょう。
router.put('/:id', function (req, res) {
    collection(collectionName).findOneAndUpdate({_id: new ObjectID(req.params.id)}, req.body, {}, function(err, r){
        res.send(r);
    })
})

// deleteメソッドを使用します。deleteとは一体何でしょうか、添付のurlを参考に調べてみましょう。
router.delete('/:id', function (req, res) {
    collection(collectionName).findOneAndDelete({ _id: new ObjectID(req.params.id) }, {}, function(err, r){
        res.send(r)
    })
})

module.exports = router;
